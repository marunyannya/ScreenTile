rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // ログイン状態判定
    function isAuthenticated() {
      return request.auth != null;
    }

    // ルーム情報
    match /artifacts/{appId}/public/data/rooms/{roomId} {
    	// 認証済みユーザーなら誰でも読める（学生がルーム存在確認に使う）
      allow read: if isAuthenticated();
      // ルームの新規作成：認証済みなら可（hostId は自分のUIDに限定）
      allow create: if isAuthenticated() && request.resource.data.hostId == request.auth.uid;
      // 更新・削除：作成者本人のみ
      allow update, delete: if isAuthenticated() && resource.data.hostId == request.auth.uid;
    }

    // 学生情報
    match /artifacts/{appId}/public/data/students/{studentId} {
      // 認証済みユーザーなら誰でも読める（教員がタイル表示に使う）
      allow read: if isAuthenticated();
      // 登録・更新・削除：自分のドキュメントのみ
      allow create, update, delete: if isAuthenticated() && request.auth.uid == studentId;
    }
  }
}
